} function updateHealthAndGrowth(state, elapsedHours) { const stats = state.stats; const poorConditions = stats.hydration < 20 || stats.nutrients < 20 || stats.cleanliness < 20 || stats.energy < 12; const goodConditions = stats.hydration >= 55 && stats.nutrients >= 55 && stats.cleanliness >= 55 && stats.health >= 55; triggerEffect('spores'); } function evaluateAttention(state, now) { const strongestNeed = getStrongestNeed(state); const attentionRules = rules.attention || {}; if (!strongestNeed) { if (state.attention.activeNeed) { addLog(state, 'Pieczarka uspokoiła się.
