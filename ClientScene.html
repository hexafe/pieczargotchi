runtime.debug.sceneQuality : ''; if (debugTier) { return profiles[debugTier]; } const state = runtime.sceneRenderQuality || { tier: getInitialSceneQualityTier(), lastFrameAt: null, slowFrames: 0, fastFrames: 0 }; const frameNow = Number(now) || 0; if (state.lastFrameAt !== null && frameNow > state.lastFrameAt) { const delta = Math.min(80, frameNow - state.lastFrameAt); if (delta > 27) { state.slowFrames += 1; state.fastFrames = Math.max(0, state.fastFrames - 3); } else if (delta < 20) { state.fastFrames += 1; state.slowFrames = Math.max(0, state.slowFrames - 2); } else { state.slowFrames = Math.max(0, state.slowFrames - 1); state.fastFrames = Math.max(0, state.fastFrames - 1); } } state.lastFrameAt = frameNow; if (state.slowFrames >= 36) { state.tier = state.tier === 'high' ? 'balanced' : 'low'; state.slowFrames = 0; state.fastFrames = 0; } else if (state.fastFrames >= 240) { state.tier = state.tier === 'low' ? 'balanced' : 'high'; state.slowFrames = 0; state.fastFrames = 0; } runtime.sceneRenderQuality = state; return profiles[state.tier] || profiles.balanced; } function getInitialSceneQualityTier() { const memory = typeof navigator !== 'undefined' ?
