function drawFrame(now) { const ctx = runtime.ctx; const wallNow = getRuntimeNow(); updateSleepVisualStart(wallNow); ctx.imageSmoothingEnabled = false; ctx.clearRect(0, 0, canvasSize, canvasSize); if ( runtime.viewMode === 'arena' && isArenaUnlockedForRuntime() && !(runtime.state && runtime.state.gameOver && runtime.state.gameOver.active) && typeof drawBattleFrame === 'function' ) { drawBattleFrame(ctx, now, wallNow); return; } drawStageBackground(ctx, now); updateImmersionReaction(wallNow); drawImmersionBackground(ctx, now, wallNow); const selectedAnimation = selectAnimation(runtime.state, wallNow); const drewAnimation = drawSelectedAnimation(ctx, selectedAnimation, wallNow); const drewHeldAnimation = !drewAnimation && drawLastReadyAnimation(ctx, selectedAnimation, wallNow); if (!drewAnimation && !drewHeldAnimation) { drawFallbackMushroom(ctx, runtime.state && runtime.state.mode === 'sleeping' ? 'critical' : 'sick'; return buildAnimationSelection( findStageAnimation(stage, recoveryState) || findStageAnimation(stage, 'sleep'), null ); } const activity = getActiveActivity(state, now); if (activity) { if (activity.type === 'wake') { return buildAnimationSelection(findStageAnimation(stage, 'wake'), getActivityStartedAt(activity, now)); } const activityAnimation = findActivityAnimation(activity.type, stage); if (activityAnimation) { return buildAnimationSelection(activityAnimation, getActivityStartedAt(activity, now)); } } if (state.mode === 'sleeping') { return buildAnimationSelection(findStageAnimation(stage, 'sleep'), null); } const need = getStrongestNeed(state); if (need) { const needState = need.severity === 'critical' ? 'critical' : getNeedAnimationState(need.id); return buildAnimationSelection( findStageAnimation(stage, needState) || findStageAnimation(stage, 'idle'), null ); } const neutralEgg = getNeutralMushroomEasterEgg(state); if (neutralEgg) { const neutralState = getNeutralEasterEggAnimationState(neutralEgg); return buildAnimationSelection( findEasterEggAnimation(stage, neutralState) || findEasterEggAnimation(stage, 'neutral') || findStageAnimation(stage, 'idle'), null ); } const immersion = getActiveImmersionReaction(now); if (immersion && immersion.state) { const immersionAnimation = findExactStageAnimation(stage, immersion.state); if (immersionAnimation) { return buildAnimationSelection(immersionAnimation, immersion.startedAt); } } if (state.stats.happiness > 82 && state.stats.health > 78 && state.patch.quality > 82) { return buildAnimationSelection(findStageAnimation(stage, 'excellent') || findStageAnimation(stage, 'happy'), null); } if (state.stats.happiness > 78 && state.stats.health > 72 && state.patch.quality > 70) { return buildAnimationSelection(findStageAnimation(stage, 'happy'), null); } return buildAnimationSelection(findStageAnimation(stage, 'idle'), null); } function getDebugAnimationSelection(stage, now) { if (!runtime.debug || !runtime.debug.enabled || !runtime.debug.forcedAnimation || runtime.debug.forcedAnimation === 'auto') { return null; } const parts = runtime.debug.forcedAnimation.split('.'); const kind = parts[0]; const name = parts[1]; if (!name) { return null; } const startedAt = runtime.debug.forcedAnimationStartedAt || now; if (kind === 'activity') { return buildAnimationSelection(findActivityAnimation(name, stage), startedAt); } return buildAnimationSelection(findStageAnimation(stage, name), startedAt); } function getActiveActivity(state, now) { const activity = state.currentActivity; if (!activity || Number(activity.until) <= now || activity.type === 'notice') { return null; } return activity; } function findStageAnimation(stage, stateName) { return animationsByKey.get(stage + '.' + stateName) || animationsByKey.get(stage + '.idle') || animationsByKey.get('spore.' + stateName) || animationsByKey.get('spore.idle') || null; } function findExactStageAnimation(stage, stateName) { return animationsByKey.get(stage + '.' + stateName) || null; } function findActivityAnimation(activityType, stage) { const animation = animationsByKey.get(stage + '.activity.' + activityType) || animationsByKey.get('activity.' + activityType); if (!animation) { return null; } if (animation.minStage && !isStageAtLeast(stage, animation.minStage)) { return null; } return animation; } function findEasterEggAnimation(stage, stateName) { return animationsByKey.get(stage + '.easter.' + stateName) || animationsByKey.get('adult.easter.' + stateName) || null; } function getNeutralEasterEggAnimationState(neutralEgg) { if (neutralEgg && neutralEgg.id === 'iwonia' && isRainyEasterEggWeather()) { return 'neutral_rain'; } return 'neutral'; } function isRainyEasterEggWeather() { const scene = getCurrentWeatherScene(); if (!scene || !['rain', 'storm'].includes(scene.condition)) { return false; } return Math.max( 0, Number(scene.rain) || 0, Number(scene.showers) || 0, Number(scene.precipitation) || 0, Number(scene.rainIntensity) || 0 ) > 0; } function buildAnimationSelection(animation, startedAt) { if (!animation) { return null; } return { animation: animation, startedAt: startedAt }; } function getActivityStartedAt(activity, now) { const startedAt = Number(activity.startedAt); if (Number.isFinite(startedAt) && startedAt > 0) { return startedAt; } const until = Number(activity.until); if (Number.isFinite(until) && until > now) { return until - (rules.activityMs || 2400); } return now; } function getNeedAnimationState(needId) { const mapping = { hydration: 'dry', nutrients: 'hungry', happiness: 'tired', cleanliness: 'dirty', energy: 'tired', health: 'sick' }; return mapping[needId] || 'idle'; } function drawSelectedAnimation(ctx, selection, now) { if (!selection || !selection.animation) { return false; } const animation = selection.animation; const image = runtime.assets[animation.key]; if (!image) { ensureAssetLoaded(animation.key); return false; } const requestedStart = selection.startedAt || (runtime.currentAnimationKey === animation.key ? runtime.currentAnimationStartedAt : now) || now; if ( runtime.currentAnimationKey !== animation.key || Math.abs((runtime.currentAnimationStartedAt || 0) - requestedStart) > 50 ) { runtime.currentAnimationKey = animation.key; runtime.currentAnimationStartedAt = requestedStart; } drawAnimationFrame(ctx, animation, image, now, runtime.currentAnimationStartedAt); return true; } function drawLastReadyAnimation(ctx, selection, now) { const selectedKey = selection && selection.animation ? selection.animation.key : null; if (!previousAnimation || !previousImage) { return false; } drawAnimationFrame(ctx, previousAnimation, previousImage, now, runtime.currentAnimationStartedAt || now); return true; } function drawAnimationFrame(ctx, animation, image, now, startedAt) { const frameWidth = animation.frameWidth || canvasSize; const frameHeight = animation.frameHeight || canvasSize; const frame = getAnimationFrame(animation, now - startedAt); const sleepMotion = getAnimationSleepMotion(animation, now); if (sleepMotion) { ctx.save(); ctx.translate(canvasSize / 2 + sleepMotion.x, canvasSize / 2 + sleepMotion.y); ctx.scale(sleepMotion.scaleX, sleepMotion.scaleY); ctx.drawImage(image, frame * frameWidth, 0, frameWidth, frameHeight, -canvasSize / 2, -canvasSize / 2, canvasSize, canvasSize); const normalSleep = state && state.mode === 'sleeping' && !(state.recovery && state.recovery.active) && !(state.gameOver && state.gameOver.active); if (!normalSleep) { runtime.sleepVisualStartedAt = 0; return; } if (!runtime.sleepVisualStartedAt) { runtime.sleepVisualStartedAt = now; } } function getAnimationSleepMotion(animation, now) { const state = runtime.state; if ( !state || state.mode !== 'sleeping' || !runtime.sleepVisualStartedAt || (state.recovery && state.recovery.active) || (state.gameOver && state.gameOver.active) || !animation || animation.kind !== 'stage' || animation.state !== 'sleep' ) { return null; } return getSleepBodyMotion(now, runtime.sleepVisualStartedAt); } function getSleepBodyMotion(now, startedAt) { const elapsed = Math.max(0, now - startedAt); const breathWave = (Math.sin(elapsed / 1850 - Math.PI / 2) + 1) / 2; const breath = smoothStep(0, 1, breathWave); const wriggleCycle = positiveModulo(elapsed + 2300, 17500) / 17500; const wriggleWindow = smoothStep(0.05, 0.13, wriggleCycle) * (1 - smoothStep(0.27, 0.42, wriggleCycle)); const wriggle = Math.sin(wriggleCycle * Math.PI * 7) * wriggleWindow; return { x: Math.round(wriggle * 3), y: Math.round(-breath * 2 + Math.sin(wriggleCycle * Math.PI * 4) * wriggleWindow), scaleX: 1 - breath * 0.003 + Math.abs(wriggle) * 0.003, scaleY: 1 + breath * 0.006, breath: Number(breath.toFixed(3)), wriggle: Number(wriggle.toFixed(3)) }; } function recordSleepBodyMotion(motion) { if (!runtime.motionDiagnostics) { return; } runtime.motionDiagnostics.sleepBody = { x: motion.x, y: motion.y, scaleX: Number(motion.scaleX.toFixed(4)), scaleY: Number(motion.scaleY.toFixed(4)), breath: motion.breath, wriggle: motion.wriggle }; } function getAnimationFrame(animation, elapsedMs) { const durations = Array.isArray(animation.frameDurationsMs) && animation.frameDurationsMs.length ? animation.frameDurationsMs : [1000 / (animation.fps || 4)]; const totalDuration = durations.reduce(function(total, duration) { return total + Math.max(1, Number(duration) || 1); }, 0); let remaining = animation.loop === false ? Math.min(Math.max(0, elapsedMs), totalDuration - 1) : positiveModulo(elapsedMs, totalDuration); for (let index = 0; index < durations.length; index += 1) { const duration = Math.max(1, Number(durations[index]) || 1); if (remaining < duration) { return Math.min(index, (animation.frameCount || durations.length) - 1); } remaining -= duration; } return Math.max(0, (animation.frameCount || durations.length) - 1); } function positiveModulo(value, modulo) { return ((value % modulo) + modulo) % modulo; } function drawNeutralEasterEggCue(ctx, now) { const neutralEgg = getNeutralMushroomEasterEgg(runtime.state); if (!neutralEgg) { return; } drawNeutralEasterEggBubble(ctx, now, neutralEgg.bubbleLines || [':|']); }); ctx.restore(); } function drawTemporaryOverlays(ctx, now, wallNow, selectedAnimation, drewAnimation) { const need = getStrongestNeed(runtime.state); const activity = getActiveActivity(runtime.state, wallNow);
