function createDefaultBattleState() { return { version: 1, mode: 'idle', unlockedAt: null, training: { strength: 0, defense: 0, speed: 0, focus: 0 }, activeBattle: null, rewards: { wins: 0, losses: 0, trophies: 0 }, log: [] }; } function normalizeBattleState(rawBattle) { const base = createDefaultBattleState(); const battle = isPlainObject(rawBattle) ? rawBattle : {}; const training = isPlainObject(battle.training) ? battle.training : {}; const rewards = isPlainObject(battle.rewards) ? battle.rewards : {}; const validModes = ['idle', 'training', 'matchSetup', 'choosingMove', 'resolvingTurn', 'animatingEvents', 'victory', 'defeat', 'rewards']; return { version: 1, mode: validModes.includes(battle.mode) ? battle.mode : base.mode, unlockedAt: battle.unlockedAt || null, training: { strength: clampInteger(training.strength, 0, 99, 0), defense: clampInteger(training.defense, 0, 99, 0), speed: clampInteger(training.speed, 0, 99, 0), focus: clampInteger(training.focus, 0, 99, 0) }, activeBattle: isPlainObject(battle.activeBattle) ? deepClone(battle.activeBattle) : null, rewards: { wins: clampInteger(rewards.wins, 0, 9999, 0), losses: clampInteger(rewards.losses, 0, 9999, 0), trophies: clampInteger(rewards.trophies, 0, 9999, 0) }, log: Array.isArray(battle.log) ? battle.log.slice(0, 8) : [] }; } function isArenaUnlocked(state, rules) { if (!isPlainObject(state)) { return false; } const battleRules = rules && rules.battle ? rules.battle : {}; const unlockStage = battleRules.unlockStage || 'legendary'; const currentStage = state.stage || ''; if (currentStage === unlockStage) { return true; } const thresholds = Array.isArray(rules && rules.stageThresholds) ? rules.stageThresholds : []; const stageOrder = thresholds.map(function(stage) { return stage.id; }); const currentIndex = stageOrder.indexOf(currentStage); const unlockIndex = stageOrder.indexOf(unlockStage); return currentIndex >= 0 && unlockIndex >= 0 && currentIndex >= unlockIndex; } function deriveBattleStats(state, battleRules) { const stats = isPlainObject(state && state.stats) ? state.stats : {}; const battle = normalizeBattleState(state && state.battle); const trainingCaps = battleRules && battleRules.trainingCaps ? battleRules.trainingCaps : {}; const training = { strength: Math.min(battle.training.strength, Number(trainingCaps.strength) || battle.training.strength), defense: Math.min(battle.training.defense, Number(trainingCaps.defense) || battle.training.defense), speed: Math.min(battle.training.speed, Number(trainingCaps.speed) || battle.training.speed), focus: Math.min(battle.training.focus, Number(trainingCaps.focus) || battle.training.focus) }; const nutrients = clampNumber(Number(stats.nutrients), 0, 100, 70); const cleanliness = clampNumber(Number(stats.cleanliness), 0, 100, 70); return { hp: Math.round(52 + health * 0.42 + hydration * 0.10 + training.defense * 1.4), stamina: Math.round(42 + energy * 0.42 + happiness * 0.10 + training.focus * 1.1), strength: Math.round(7 + nutrients * 0.10 + training.strength * 1.3), attack: Math.round(8 + nutrients * 0.10 + happiness * 0.04 + training.strength * 1.25), defense: Math.round(7 + cleanliness * 0.09 + health * 0.05 + training.defense * 1.2), speed: Math.round(7 + energy * 0.10 + hydration * 0.04 + training.speed * 1.25), focus: Math.round(7 + happiness * 0.09 + health * 0.04 + training.focus * 1.25) }; } function trainBattleStat(rawState, statId, rules, now) { const state = isPlainObject(rawState) ? deepClone(rawState) : {}; const battleRules = rules && rules.battle ? rules.battle : {}; const caps = battleRules.trainingCaps || {}; const validStats = ['strength', 'defense', 'speed', 'focus']; const cost = Math.max(1, clampInteger(battleRules.trainingCost, 1, 999, 1)); state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); state.battle = normalizeBattleState(state.battle); if (!isArenaUnlocked(state, rules)) { return { ok: false, reason: 'Arena jest zablokowana.', state: state }; } if (!validStats.includes(statId)) { return { ok: false, reason: 'Nieznany trening.', state: state }; } const cap = Math.max(0, clampInteger(caps[statId], 0, 999, 20)); if (state.battle.training[statId] >= cap) { return { ok: false, reason: 'Ten trening ma już limit.', state: state }; } const availableSpores = Math.max(Number(state.inventory.spores) || 0, Number(state.coins) || 0); if (availableSpores < cost) { return { ok: false, reason: 'Brakuje zarodników.', state: state }; } const remaining = availableSpores - cost; state.inventory.spores = remaining; state.coins = remaining; state.battle.training[statId] += 1; state.battle.mode = 'training'; addBattleLog(state.battle, 'Trening: ' + statId + ' +' + 1 + '.', now, rules); return { ok: true, state: state }; } function startBattle(rawState, rules, now, seedInput) { const state = isPlainObject(rawState) ? deepClone(rawState) : {}; const battleRules = rules && rules.battle ? rules.battle : {}; const timestamp = Number(now) || Date.now(); const seed = clampInteger(seedInput || buildBattleSeed(state, timestamp), 1, 2147483646, 1); } function resolveBattleRound(activeBattle, playerMoveId, moveCatalog) { const opponentSelection = chooseOpponentMove(activeBattle, moveCatalog); const battleWithSelectionSeed = Object.assign({}, activeBattle, { rngSeed: opponentSelection.rngSeed }); const resolved = resolveBattleTurn( battleWithSelectionSeed, playerMoveId, opponentSelection.moveId, moveCatalog ); return Object.assign({}, resolved, { selectedMoves: { player: playerMoveId, opponent: opponentSelection.moveId } }); } function chooseOpponentMove(activeBattle, moveCatalog) { const catalog = indexMoves(moveCatalog); const moves = Object.keys(catalog).map(function(key) { return catalog[key]; }); const opponent = normalizeCombatant(activeBattle && activeBattle.opponent); const viable = moves.filter(function(move) { return opponent.stamina >= clampInteger(move.staminaCost, 0, 999, 0); }); const roll = nextRandom(activeBattle && activeBattle.rngSeed); const candidates = viable.length ? viable : moves; const selected = chooseWeightedOpponentMove(candidates, opponent, roll.value) || firstMove(catalog); return { moveId: selected.id, rngSeed: roll.seed }; } function chooseWeightedOpponentMove(candidates, opponent, value) { const weights = isPlainObject(opponent.weights) ? opponent.weights : {}; const hpRatio = opponent.maxHp > 0 ? opponent.hp / opponent.maxHp : 1; const weighted = candidates.map(function(move) { let weight = Math.max(1, Number(weights[move.id]) || 1); if (hpRatio < 0.34 && move.guard) { weight += 3; } if (opponent.stamina < 16 && move.staminaCost <= 8) { weight += 2; } if (opponent.strategy === 'speed' && move.stat === 'speed') { weight += 2; } if (opponent.strategy === 'defensive' && move.guard) { weight += 2; } return { move: move, weight: weight }; }); const total = weighted.reduce(function(sum, item) { return sum + item.weight; }, 0); let cursor = value * total; for (let index = 0; index < weighted.length; index += 1) { cursor -= weighted[index].weight; if (cursor <= 0) { return weighted[index].move; } } return weighted.length ? weighted[weighted.length - 1].move : null; } function applyBattleOutcomeRewards(rawState, rules, now) { const state = isPlainObject(rawState) ? deepClone(rawState) : {}; const battleRules = rules && rules.battle ? rules.battle : {}; const victoryRewards = battleRules.victoryRewards || { spores: 2, wins: 1, trophies: 1 }; const defeatRewards = battleRules.defeatRewards || { losses: 1 }; const battle = normalizeBattleState(state.battle); const activeBattle = isPlainObject(battle.activeBattle) ? deepClone(battle.activeBattle) : null; state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); state.battle = battle; if (!activeBattle || activeBattle.rewarded || !['victory', 'defeat'].includes(activeBattle.mode)) { return { ok: false, state: state }; } if (activeBattle.mode === 'victory') { const spores = Math.max(0, clampInteger(victoryRewards.spores, 0, 999, 2)); state.battle.rewards.wins += Math.max(0, clampInteger(victoryRewards.wins, 0, 99, 1)); state.battle.rewards.trophies += Math.max(0, clampInteger(victoryRewards.trophies, 0, 99, 1)); state.inventory.spores = (Number(state.inventory.spores) || 0) + spores; state.coins = (Number(state.coins) || 0) + spores; addBattleLog(state.battle, 'Wygrana: +' + spores + ' zarodniki i trofeum.', now, rules); } else { state.battle.rewards.losses += Math.max(0, clampInteger(defeatRewards.losses, 0, 99, 1)); addBattleLog(state.battle, 'Porażka zapisana bez kary dla opieki.', now, rules); } activeBattle.rewarded = true; state.battle.mode = activeBattle.mode; state.battle.activeBattle = activeBattle; return { ok: true, state: state }; } function resolveBattleTurn(activeBattle, playerMoveId, opponentMoveId, moveCatalog) { const battle = isPlainObject(activeBattle) ? deepClone(activeBattle) : {}; const catalog = indexMoves(moveCatalog); const playerMove = catalog[playerMoveId] || catalog.sporeJab || firstMove(catalog); const opponentMove = catalog[opponentMoveId] || catalog.sporeJab || firstMove(catalog); const player = normalizeCombatant(battle.player); const opponent = normalizeCombatant(battle.opponent); let seed = clampInteger(battle.rngSeed, 1, 2147483646, 1); const events = []; const playerSpeed = getEffectiveBattleSpeed(player); const opponentSpeed = getEffectiveBattleSpeed(opponent); const order = playerSpeed >= opponentSpeed ? [ { actor: 'player', source: player, target: opponent, move: playerMove }, { actor: 'opponent', source: opponent, target: player, move: opponentMove } ] : [ { actor: 'opponent', source: opponent, target: player, move: opponentMove }, { actor: 'player', source: player, target: opponent, move: playerMove } ]; order.forEach(function(turn) { if (player.hp <= 0 || opponent.hp <= 0) { return; } const outcome = applyBattleMove(turn.source, turn.target, turn.move, seed); seed = outcome.rngSeed; events.push(Object.assign({ actor: turn.actor, moveId: turn.move.id }, outcome.event)); }); const mode = opponent.hp <= 0 ? 'victory' : player.hp <= 0 ? 'defeat' : 'choosingMove'; tickCombatantStatusEffects(player); tickCombatantStatusEffects(opponent); return Object.assign({}, battle, { mode: mode, rngSeed: seed, player: player, opponent: opponent, turn: clampInteger(battle.turn, 0, 9999, 0) + 1, events: events }); } function indexMoves(moveCatalog) { const fallback = [ { id: 'sporeJab', staminaCost: 8, power: 14, accuracy: 0.94, stat: 'strength' }, { id: 'capGuard', staminaCost: 6, power: 4, accuracy: 1, stat: 'defense', guard: 0.35 } ]; return (Array.isArray(moveCatalog) && moveCatalog.length ? moveCatalog : fallback).reduce(function(result, move) { result[move.id] = move; return result; }, {}); } function firstMove(catalog) { const firstKey = Object.keys(catalog)[0]; return catalog[firstKey]; } function normalizeCombatant(rawCombatant) { const combatant = isPlainObject(rawCombatant) ? rawCombatant : {}; const hp = clampInteger(combatant.hp, 0, 999, 80); const stamina = clampInteger(combatant.stamina, 0, 999, 50); return { name: combatant.name || '', hp: hp, maxHp: clampInteger(combatant.maxHp, Math.max(1, hp), 999, Math.max(1, hp)), stamina: stamina, maxStamina: clampInteger(combatant.maxStamina, Math.max(1, stamina), 999, Math.max(1, stamina)), strength: clampInteger(combatant.strength, 0, 999, 10), attack: clampInteger(combatant.attack, 0, 999, 12), defense: clampInteger(combatant.defense, 0, 999, 10), speed: clampInteger(combatant.speed, 0, 999, 10), focus: clampInteger(combatant.focus, 0, 999, 10), guard: clampNumber(Number(combatant.guard), 0, 0.9, 0), statusEffects: normalizeBattleStatusEffects(combatant.statusEffects), strategy: combatant.strategy || 'balanced', weights: isPlainObject(combatant.weights) ? deepClone(combatant.weights) : {} }; } function normalizeBattleStatusEffects(statusEffects) { const source = isPlainObject(statusEffects) ? statusEffects : {}; const result = {}; Object.keys(source).forEach(function(key) { const item = isPlainObject(source[key]) ? source[key] : {}; const turns = clampInteger(item.turns, 0, 99, 0); if (turns > 0) { result[key] = { turns: turns, value: clampInteger(item.value, 0, 999, 0) }; } }); return result; } function getEffectiveBattleSpeed(combatant) { const slow = combatant.statusEffects && combatant.statusEffects.slow; return Math.max(0, combatant.speed - (slow ? Number(slow.value) || 0 : 0)); } function tickCombatantStatusEffects(combatant) { const statusEffects = normalizeBattleStatusEffects(combatant.statusEffects); Object.keys(statusEffects).forEach(function(key) { statusEffects[key].turns -= 1; if (statusEffects[key].turns <= 0) { delete statusEffects[key]; } }); combatant.statusEffects = statusEffects; } function applyBattleMove(source, target, move, seed) { const cost = clampInteger(move.staminaCost, 0, 999, 0); if (source.stamina < cost) { source.stamina = Math.min(999, source.stamina + 6); return { rngSeed: seed, event: { type: 'rest', damage: 0 } }; } source.stamina = Math.max(0, source.stamina - cost); const roll = nextRandom(seed); const accuracy = clampNumber(Number(move.accuracy), 0, 1, 1); if (roll.value > accuracy) { return { rngSeed: roll.seed, event: { type: 'miss', damage: 0 } }; } const statBonus = move.stat && source[move.stat] ? move.selfEffect.type : null, statusEffect: isPlainObject(move.statusEffect) ? move.statusEffect.type : null } }; } function applyBattleSelfEffect(source, effect) { if (!isPlainObject(effect)) { return; } if (effect.type === 'stamina') { source.stamina = Math.min(source.maxStamina, source.stamina + clampInteger(effect.value, 0, 999, 0)); } if (effect.type === 'heal') { source.hp = Math.min(source.maxHp, source.hp + clampInteger(effect.value, 0, 999, 0)); } } function applyBattleStatusEffect(source, target, effect) { if (!isPlainObject(effect) || !effect.type) { return; } const receiver = effect.target === 'self' ? state.battle.rewards : {}; const base = Math.floor(Number(now) || 1) + (Number(rewards.wins) || 0) * 9973 + (Number(rewards.losses) || 0) * 3181 + (Number(rewards.trophies) || 0) * 7919; return positiveModulo(base, 2147483646) + 1; } function createBattleOpponent(state, battleRules, seed) { const rewards = state && state.battle && state.battle.rewards ?
