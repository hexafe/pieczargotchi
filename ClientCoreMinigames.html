function createMinigameSessionCore(minigameId, rules, now, seedInput, rawState, scene) { const minigames = rules && rules.minigames ? rules.minigames : {}; const config = minigames[minigameId]; const timestamp = Number(now) || Date.now(); if (!config) { return { ok: false, reason: 'Nieznana minigra.' }; } if (config.requiresStage && rawState && !isGameplayStageAtLeastCore(rawState.stage, config.requiresStage)) { return { ok: false, reason: 'To wyzwanie czeka na legendarną Pieczarkę.' }; } const seed = clampInteger(seedInput || timestamp, 1, 2147483646, 1); const habitat = rawState ? getHabitatInfluenceCore(rawState, rules) : { tags: {} }; return { ok: true, session: { id: minigameId, label: config.label || 'Minigra', seed: seed, startedAt: timestamp, until: timestamp + (Number(config.durationMs) || 20000), score: 0, caught: [], mistakes: 0, habitatTags: Object.assign({}, habitat.tags || {}), sceneTone: scene && scene.dayPhase ? scene.dayPhase : null } }; } function finishMinigameCore(rawState, session, rules, now) { const state = normalizeProgressionState(rawState, rules); const minigames = rules && rules.minigames ? rules.minigames : {}; const config = session && minigames[session.id]; const timestamp = Number(now) || Date.now(); if (!config) { return { ok: false, reason: 'Nieznana minigra.', state: state }; } if (config.requiresStage && !isGameplayStageAtLeastCore(state.stage, config.requiresStage)) { state.minigames = { active: null, lastResult: state.minigames && state.minigames.lastResult || null }; return { ok: false, reason: 'To wyzwanie czeka na legendarną Pieczarkę.', state: state }; } const score = clampInteger(session.score, 0, 999, 0); const bestCombo = clampInteger(session.bestCombo, 0, 999, 0); const mistakes = clampInteger(session.mistakes, 0, 999, 0); const previousBest = state.history && state.history.minigames && state.history.minigames[session.id] ? clampInteger(state.history.minigames[session.id].bestScore, 0, 999999, 0) : 0; const targetScore = getMinigameTargetScoreCore(session.id, config); const rewards = calculateMinigameRewardsCore(config, score); state.stats = Object.assign({}, state.stats || {}); Object.keys(rewards.stats).forEach(function(stat) { state.stats[stat] = clampNumber(Number(state.stats[stat]) + rewards.stats[stat], 0, 100, 0); }); if (rewards.spores) { state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); state.inventory.spores = (Number(state.inventory.spores) || 0) + rewards.spores; state.coins = (Number(state.coins) || 0) + rewards.spores; } if (rewards.inventory) { state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); Object.keys(rewards.inventory).forEach(function(itemId) { state.inventory[itemId] = (Number(state.inventory[itemId]) || 0) + rewards.inventory[itemId]; }); state.coins = Number(state.inventory.spores) || Number(state.coins) || 0; } state.history = normalizeHistoryCore(state.history); const item = state.history.minigames[session.id] || { plays: 0, bestScore: 0, lastScore: 0, lastPlayedAt: null }; item.plays += 1; item.lastScore = score; item.bestScore = Math.max(item.bestScore, score); item.lastPlayedAt = new Date(timestamp).toISOString(); state.history.minigames[session.id] = item; state.minigames = { active: null, lastResult: { id: session.id, score: score, previousBest: previousBest, newBest: score > previousBest, bestCombo: bestCombo, mistakes: mistakes, targetScore: targetScore, tier: getMinigameResultTierCore(score, targetScore, mistakes), rewards: rewards, endedAt: new Date(timestamp).toISOString() } }; return { ok: true, state: state, rewards: rewards }; } function calculateMinigameRewardsCore(config, score) { const rewardRules = config.rewards || {}; const stats = {}; const inventory = {}; const hydrationPerPoint = Number(rewardRules.hydrationPerCatch) || Number(rewardRules.hydrationPerPoint) || 0; const happinessPerPoint = Number(rewardRules.happinessPerCatch) || Number(rewardRules.happinessPerPop) || Number(rewardRules.happinessPerPoint) || 0; const nutrientsPerPoint = Number(rewardRules.nutrientsPerPoint) || 0; const cleanlinessPerPoint = Number(rewardRules.cleanlinessPerPoint) || 0; const energyPerPoint = Number(rewardRules.energyPerPoint) || 0; if (hydrationPerPoint || rewardRules.hydrationBase || rewardRules.hydrationMax) { stats.hydration = Math.min( Number(rewardRules.hydrationMax) || 18, (Number(rewardRules.hydrationBase) || 0) + score * hydrationPerPoint ); } if (happinessPerPoint || rewardRules.happinessBase || rewardRules.happinessMax) { stats.happiness = Math.min( Number(rewardRules.happinessMax) || 6, (Number(rewardRules.happinessBase) || 0) + score * happinessPerPoint ); } if (nutrientsPerPoint || rewardRules.nutrientsBase || rewardRules.nutrientsMax) { stats.nutrients = Math.min( Number(rewardRules.nutrientsMax) || 10, (Number(rewardRules.nutrientsBase) || 0) + score * nutrientsPerPoint ); } if (cleanlinessPerPoint || rewardRules.cleanlinessBase || rewardRules.cleanlinessMax) { stats.cleanliness = Math.min( Number(rewardRules.cleanlinessMax) || 4, (Number(rewardRules.cleanlinessBase) || 0) + score * cleanlinessPerPoint ); } if (energyPerPoint || rewardRules.energyBase || rewardRules.energyMax) { stats.energy = Math.min( Number(rewardRules.energyMax) || 3, (Number(rewardRules.energyBase) || 0) + score * energyPerPoint ); } if (rewardRules.substratePerPoint || rewardRules.substrateMax) { inventory.substrate = Math.max(0, Math.min( Number(rewardRules.substrateMax) || 1, Math.floor(score * (Number(rewardRules.substratePerPoint) || 0)) )); } return { stats: removeZeroDeltas(stats), inventory: removeZeroDeltas(inventory), spores: calculateMinigameSporeRewardCore(rewardRules, score) }; } function getMinigameTargetScoreCore(minigameId, config) { const source = config || {}; if (Number(source.masteryTarget)) { return Math.max(1, Math.floor(Number(source.masteryTarget))); } if (minigameId === 'dewCatch') { return Math.max(1, Math.floor(Number(source.dropCount) || 24)); } if (minigameId === 'sporePop') { return Math.max(1, Math.floor(Number(source.targetCount) || 20)); } if (minigameId === 'compostSort') { return Math.max(1, Math.floor(Number(source.pieceCount) || 18)); } if (minigameId === 'rhythmHum') { return Math.max(1, Math.floor((Number(source.beatCount) || 8) * 3)); } return 10; } function getMinigameResultTierCore(score, targetScore, mistakes) { const target = Math.max(1, Number(targetScore) || 1); const ratio = Math.max(0, Number(score) || 0) / target; if (ratio >= 1 && !mistakes) { return 'perfect'; } if (ratio >= 1) { return 'clear'; } if (ratio >= 0.72) { return 'near'; } return 'practice'; } function judgeRhythmInputCore(expectedLane, inputLane, deltaMs, windows, offsetMs) { const timing = windows || {}; const offset = Number(offsetMs) || 0; const delta = (Number(deltaMs) || 0) + offset; const abs = Math.abs(delta); const ok = Math.max(110, Number(timing.ok) || 190); const good = Math.max(75, Number(timing.good) || 125); const perfect = Math.max(40, Number(timing.perfect) || 70); const miss = Math.max(ok, Number(timing.miss) || 260); const expected = String(expectedLane || ''); const input = String(inputLane || ''); if (!expected) { return { judgment: 'miss', points: 0, timing: 'stray', delta: delta, expectedLane: expected, inputLane: input }; } if (input !== expected) { return { judgment: 'bad', points: 0, timing: 'wrongLane', delta: delta, expectedLane: expected, inputLane: input }; } if (abs > miss) { return { judgment: 'miss', points: 0, timing: delta < 0 ? 'early' : 'late', delta: delta, expectedLane: expected, inputLane: input }; } if (abs > ok) { return { judgment: 'miss', points: 0, timing: delta < 0 ? 'early' : 'late', delta: delta, expectedLane: expected, inputLane: input }; } if (abs <= perfect) { return { judgment: 'perfect', points: 3, timing: 'perfect', delta: delta, expectedLane: expected, inputLane: input }; } if (abs <= good) { return { judgment: 'good', points: 2, timing: delta < 0 ? 'slightlyEarly' : 'slightlyLate', delta: delta, expectedLane: expected, inputLane: input }; } return { judgment: 'ok', points: 1, timing: delta < 0 ? 'earlyOk' : 'lateOk', delta: delta, expectedLane: expected, inputLane: input }; } function calculateMinigameSporeRewardCore(rewardRules, score) { if (rewardRules.sporesPerPop || rewardRules.sporesPerPoint || rewardRules.sporesMax) { const perPoint = Number(rewardRules.sporesPerPop) || Number(rewardRules.sporesPerPoint) || 0; const max = Number(rewardRules.sporesMax) || 999; return Math.max(0, Math.min(max, Math.floor(score * perPoint))); } return Math.max(0, Math.floor(score / 12)); }
