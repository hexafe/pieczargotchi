function createMinigameSessionCore(minigameId, rules, now, seedInput, rawState, scene) { const minigames = rules && rules.minigames ? rules.minigames : {}; const config = minigames[minigameId]; const timestamp = Number(now) || Date.now(); if (!config) { return { ok: false, reason: 'Nieznana minigra.' }; } const seed = clampInteger(seedInput || timestamp, 1, 2147483646, 1); const habitat = rawState ? getHabitatInfluenceCore(rawState, rules) : { tags: {} }; return { ok: true, session: { id: minigameId, label: config.label || 'Minigra', seed: seed, startedAt: timestamp, until: timestamp + (Number(config.durationMs) || 20000), score: 0, caught: [], mistakes: 0, habitatTags: Object.assign({}, habitat.tags || {}), sceneTone: scene && scene.dayPhase ? scene.dayPhase : null } }; } function finishMinigameCore(rawState, session, rules, now) { const state = normalizeProgressionState(rawState, rules); const minigames = rules && rules.minigames ? rules.minigames : {}; const config = session && minigames[session.id]; const timestamp = Number(now) || Date.now(); if (!config) { return { ok: false, reason: 'Nieznana minigra.', state: state }; } const score = clampInteger(session.score, 0, 999, 0); const rewards = calculateMinigameRewardsCore(config, score); state.stats = Object.assign({}, state.stats || {}); Object.keys(rewards.stats).forEach(function(stat) { state.stats[stat] = clampNumber(Number(state.stats[stat]) + rewards.stats[stat], 0, 100, 0); }); if (rewards.spores) { state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); state.inventory.spores = (Number(state.inventory.spores) || 0) + rewards.spores; state.coins = (Number(state.coins) || 0) + rewards.spores; } if (rewards.inventory) { state.inventory = Object.assign({ spores: 0 }, state.inventory || {}); Object.keys(rewards.inventory).forEach(function(itemId) { state.inventory[itemId] = (Number(state.inventory[itemId]) || 0) + rewards.inventory[itemId]; }); state.coins = Number(state.inventory.spores) || Number(state.coins) || 0; } state.history = normalizeHistoryCore(state.history); const item = state.history.minigames[session.id] || { plays: 0, bestScore: 0, lastScore: 0, lastPlayedAt: null }; item.plays += 1; item.lastScore = score; item.bestScore = Math.max(item.bestScore, score); item.lastPlayedAt = new Date(timestamp).toISOString(); state.history.minigames[session.id] = item; state.minigames = { active: null, lastResult: { id: session.id, score: score, rewards: rewards, endedAt: new Date(timestamp).toISOString() } }; return { ok: true, state: state, rewards: rewards }; } function calculateMinigameRewardsCore(config, score) { const rewardRules = config.rewards || {}; const stats = {}; const inventory = {}; const hydrationPerPoint = Number(rewardRules.hydrationPerCatch) || Number(rewardRules.hydrationPerPoint) || 0; const happinessPerPoint = Number(rewardRules.happinessPerCatch) || Number(rewardRules.happinessPerPop) || Number(rewardRules.happinessPerPoint) || 0; const nutrientsPerPoint = Number(rewardRules.nutrientsPerPoint) || 0; const cleanlinessPerPoint = Number(rewardRules.cleanlinessPerPoint) || 0; const energyPerPoint = Number(rewardRules.energyPerPoint) || 0; if (hydrationPerPoint || rewardRules.hydrationBase || rewardRules.hydrationMax) { stats.hydration = Math.min( Number(rewardRules.hydrationMax) || 18, (Number(rewardRules.hydrationBase) || 0) + score * hydrationPerPoint ); } if (happinessPerPoint || rewardRules.happinessBase || rewardRules.happinessMax) { stats.happiness = Math.min( Number(rewardRules.happinessMax) || 6, (Number(rewardRules.happinessBase) || 0) + score * happinessPerPoint ); } if (nutrientsPerPoint || rewardRules.nutrientsBase || rewardRules.nutrientsMax) { stats.nutrients = Math.min( Number(rewardRules.nutrientsMax) || 10, (Number(rewardRules.nutrientsBase) || 0) + score * nutrientsPerPoint ); } if (cleanlinessPerPoint || rewardRules.cleanlinessBase || rewardRules.cleanlinessMax) { stats.cleanliness = Math.min( Number(rewardRules.cleanlinessMax) || 4, (Number(rewardRules.cleanlinessBase) || 0) + score * cleanlinessPerPoint ); } if (energyPerPoint || rewardRules.energyBase || rewardRules.energyMax) { stats.energy = Math.min( Number(rewardRules.energyMax) || 3, (Number(rewardRules.energyBase) || 0) + score * energyPerPoint ); } if (rewardRules.substratePerPoint || rewardRules.substrateMax) { inventory.substrate = Math.max(0, Math.min( Number(rewardRules.substrateMax) || 1, Math.floor(score * (Number(rewardRules.substratePerPoint) || 0)) )); } return { stats: removeZeroDeltas(stats), inventory: removeZeroDeltas(inventory), spores: calculateMinigameSporeRewardCore(rewardRules, score) }; } function calculateMinigameSporeRewardCore(rewardRules, score) { if (rewardRules.sporesPerPop || rewardRules.sporesPerPoint || rewardRules.sporesMax) { const perPoint = Number(rewardRules.sporesPerPop) || Number(rewardRules.sporesPerPoint) || 0; const max = Number(rewardRules.sporesMax) || 999; return Math.max(0, Math.min(max, Math.floor(score * perPoint))); } return Math.max(0, Math.floor(score / 12)); }
