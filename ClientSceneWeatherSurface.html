function drawWeatherForeground(ctx, now) { const scene = getCurrentWeatherScene(); const surface = getWeatherSurfaceState(scene, now); if (surface.snowCover > 0.04) { drawSnowBaseOverlay(ctx, scene, now, surface.snowCover); } const fogDensity = getFogVisualDensity(scene); if (fogDensity > 0.08) { drawFogForeground(ctx, scene, now, fogDensity); } } function getWeatherSurfaceState(scene, now) { const source = scene || {}; const fallbackWetness = ['rain', 'storm'].includes(source.condition) ? clamp(Number(source.rainIntensity), 0, 1, 0) : source.condition === 'fog' ? 0.42 : 0; const fallbackSnow = source.condition === 'snow' ? clamp(Number(source.snowIntensity), 0, 1, 0) : 0; const targetWetness = clamp(Number(source.surfaceWetnessTarget), 0, 1, fallbackWetness); const targetSnow = clamp(Number(source.snowCoverTarget), 0, 1, fallbackSnow); const temperature = source.temperature === null || source.temperature === undefined ? null : Number(source.temperature); const windDrying = clamp(Number(source.windLevel), 0, 1, 0) * 0.012; const heatDrying = Number.isFinite(temperature) ? clamp((temperature - 12) / 28, 0, 1, 0) * 0.010 : 0.003; const snowMelt = Number.isFinite(temperature) ? clamp((temperature - 0.5) / 8, 0, 1, 0) * 0.014 : 0.002; if (!runtime.weatherSurface) { runtime.weatherSurface = { wetness: targetWetness, snowCover: targetSnow, updatedAt: now }; } const state = runtime.weatherSurface; const elapsed = clamp((now - (state.updatedAt || now)) / 1000, 0, 2.5, 0); state.updatedAt = now; state.wetness = approachWeatherValue(state.wetness, targetWetness, elapsed, 0.18, 0.018 + windDrying + heatDrying); state.snowCover = approachWeatherValue(state.snowCover, targetSnow, elapsed, 0.12, 0.010 + snowMelt + windDrying * 0.45); return { wetness: clamp(state.wetness, 0, 1, 0), snowCover: clamp(state.snowCover, 0, 1, 0) }; } function approachWeatherValue(current, target, elapsed, riseRate, fallRate) { const value = clamp(Number(current), 0, 1, 0); const goal = clamp(Number(target), 0, 1, 0); const rate = goal >= value ? riseRate : fallRate; const step = Math.max(0, Number(elapsed) || 0) * rate; if (Math.abs(goal - value) <= step) { return goal; } return value + (goal > value ? step : -step); } function drawSnowGroundCover(ctx, scene, now, groundTop, cover) { const intensity = Math.max(0.18, cover || scene.snowIntensity || 0.25); ctx.save();
