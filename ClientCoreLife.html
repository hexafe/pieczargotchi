} function getInsectTemperatureFactor(scene) { const temperature = readAmbientTemperature(scene); if (temperature === null) { return 0.84; } if (temperature <= 4) { return 0.02; } if (temperature < 10) { return 0.08 + (temperature - 4) / 6 * 0.28;
