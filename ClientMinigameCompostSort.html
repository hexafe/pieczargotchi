} function handleCompostSortPointer(event) { if (!runtime.minigame || runtime.minigame.session.id !== 'compostSort' || !dom.compostSortCanvas) { return; } const canvas = dom.compostSortCanvas; const rect = canvas.getBoundingClientRect(); }); if (!picked) { return; } event.preventDefault(); session.caught.push(picked.id); if (picked.good) { session.score += Number(picked.points) || 1; session.combo = (Number(session.combo) || 0) + 1; session.bestCombo = Math.max(Number(session.bestCombo) || 0, Number(session.combo) || 0);
