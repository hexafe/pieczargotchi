function drawBattleFrame(ctx, now, wallNow) { runtime.currentAnimationKey = 'battle.scene'; drawBattleArenaBackground(ctx, now); const active = runtime.state && runtime.state.battle ? runtime.state.battle.activeBattle : null; if (!active) { runtime.battlePresentation = null; drawBattleSprite(ctx, 'playerLegendary', 'idle', 158, 178, 196, false, wallNow); drawBattleRing(ctx, 256, 385, 86, '#f0c45b'); return; } const presentation = getBattlePresentation(active, now); const playerOffset = getBattlePresentationOffset(presentation, 'player', now); const opponentOffset = getBattlePresentationOffset(presentation, 'opponent', now); drawBattleRing(ctx, 154, 398, 86, '#b6d66f'); drawBattleRing(ctx, 360, 398, 82, '#d8646f'); drawBattleSprite( ctx, active.player && active.player.visualId || 'playerLegendary', presentation.playerPose, 54 + playerOffset.x, 196 + playerOffset.y, 194, false, wallNow ); drawBattleSprite( ctx, active.opponent && active.opponent.visualId || 'sproutling', presentation.opponentPose, 270 + opponentOffset.x, 205 + opponentOffset.y, 184, true, wallNow ); drawBattleImpactEffects(ctx, presentation, now); } function drawBattleArenaBackground(ctx, now) { const backgroundKey = 'battle.arena.background'; const background = runtime.assets[backgroundKey]; if (background) { touchRuntimeAsset(backgroundKey); ctx.drawImage(background, 0, 0, canvasSize, canvasSize); return; } ensureAssetLoaded(backgroundKey); const pulse = Math.sin(now / 900) * 4; ctx.fillStyle = '#bedcbd'; ctx.fillRect(0, 0, canvasSize, canvasSize); ctx.fillStyle = '#97b87e'; ctx.fillRect(0, 326, canvasSize, 186); ctx.fillStyle = '#678948'; ctx.fillRect(0, 386, canvasSize, 126); ctx.fillStyle = 'rgba(255, 253, 247, 0.28)'; ctx.fillRect(58, 74 + pulse, 88, 14); ctx.fillRect(352, 108 - pulse, 92, 14); ctx.strokeStyle = '#3c2b20'; ctx.lineWidth = 4; ctx.strokeRect(42, 318, 428, 132); ctx.strokeStyle = 'rgba(60, 43, 32, 0.22)'; ctx.lineWidth = 2; ctx.beginPath(); ctx.moveTo(68, 382); ctx.lineTo(444, 382); ctx.moveTo(256, 318); ctx.lineTo(256, 450); ctx.stroke(); } function drawBattleRing(ctx, x, y, radius, color) { ctx.save(); ctx.fillStyle = color; ctx.globalAlpha = 0.32; ctx.beginPath(); ctx.ellipse(x, y, radius, 22, 0, 0, Math.PI * 2); ctx.fill(); ctx.restore(); } function drawBattleSprite(ctx, visualId, pose, x, y, size, flip) { const assetKey = getBattleSpriteAssetKey(visualId); const image = runtime.assets[assetKey]; ctx.save(); if (flip) { ctx.translate(x + size, y); ctx.scale(-1, 1); x = 0; y = 0; } if (image) { touchRuntimeAsset(assetKey); const frame = getBattlePoseFrameIndex(pose); ctx.drawImage( image, frame * 256, 0, 256, 256, x, y, size, size ); ctx.restore(); return; } ensureAssetLoaded(assetKey); drawBattleLoadingSilhouette(ctx, x, y, size); ctx.restore(); } function getBattleSpriteAssetKey(visualId) { const id = String(visualId || '').replace(/[^0-9A-Za-z_-]+/g, ''); if (id === 'playerLegendary' || id === 'legendary') { return 'battle.player.legendary'; } if (id === 'windcap') { return 'battle.opponent.windcap'; } if (id === 'eldercap') { return 'battle.opponent.eldercap'; } return 'battle.opponent.sproutling'; } function getBattlePoseFrameIndex(pose) { const poses = ['idle', 'attack', 'guard', 'hurt']; const index = poses.indexOf(pose); return index >= 0 ? index : 0; } function warmUpBattleAssets() { const keys = [ 'battle.arena.background', 'battle.player.legendary', 'battle.opponent.sproutling', 'battle.opponent.windcap', 'battle.opponent.eldercap' ]; if (runtime.battleAssetWarmup) { return runtime.battleAssetWarmup; } runtime.battleAssetWarmup = Promise.all(keys.map(function(key) { return ensureAssetLoaded(key); })).finally(function() { runtime.battleAssetWarmup = null; }); return runtime.battleAssetWarmup; } function drawBattleLoadingSilhouette(ctx, x, y, size) { const unit = size / 16; ctx.globalAlpha = 0.32; ctx.fillStyle = '#40583e'; ctx.fillRect(x + unit * 5, y + unit * 7, unit * 6, unit * 7); ctx.fillStyle = '#6c8157'; ctx.fillRect(x + unit * 2, y + unit * 3, unit * 12, unit * 4); ctx.fillRect(x + unit * 4, y + unit * 2, unit * 8, unit * 3); } function getBattlePose(active, actor) { const events = Array.isArray(active.events) ? active.events : []; const actorEvent = events.slice().reverse().find(function(event) { return event.actor === actor; }); const incomingHit = events.slice().reverse().find(function(event) { return event.actor !== actor && event.type === 'hit'; }); if (incomingHit) { return 'hurt'; } if (!actorEvent) { return 'idle'; } if (actorEvent.type === 'guard' || actorEvent.type === 'rest') { return 'guard'; } if (actorEvent.type === 'hit' || actorEvent.type === 'miss') { return 'attack'; } return 'idle'; } function getBattlePresentation(active, now) { const frameNow = Math.max(0, Number(now) || 0); const events = Array.isArray(active && active.events) ? active.events : []; const signature = getBattlePresentationSignature(active); let state = runtime.battlePresentation; if (!state || state.signature !== signature || frameNow < Number(state.startedAt)) { state = { signature: signature, startedAt: frameNow, phase: 'idle', eventIndex: -1, playerPose: 'idle', opponentPose: 'idle' }; runtime.battlePresentation = state; } const eventDuration = 720; const elapsed = Math.max(0, frameNow - state.startedAt); const eventIndex = Math.floor(elapsed / eventDuration); const event = eventIndex < events.length ? events[eventIndex] : null; const localElapsed = event ? elapsed - eventIndex * eventDuration : eventDuration; const phase = getBattlePresentationPhase(event, localElapsed); const poses = getBattlePresentationPoses(event, phase); state.phase = phase; state.eventIndex = event ? eventIndex : -1; state.playerPose = poses.player; state.opponentPose = poses.opponent; state.event = event; state.phaseProgress = getBattlePhaseProgress(phase, localElapsed); return state; } function getBattlePresentationSignature(active) { const events = Array.isArray(active && active.events) ? active.events : []; const eventKey = events.map(function(event) { return [event.actor, event.type, event.moveId, Number(event.damage) || 0].join(':'); }).join('|'); return [Number(active && active.turn) || 0, active && active.mode || 'idle', eventKey].join('#'); } function getBattlePresentationPhase(event, localElapsed) { if (!event) { return 'idle'; } if (event.type === 'guard' || event.type === 'rest') { return localElapsed < 420 ? 'guard' : 'idle'; } if (localElapsed < 190) { return 'attack'; } if (event.type === 'hit' && localElapsed < 310) { return 'impact'; } if (event.type === 'hit' && localElapsed < 520) { return 'hurt'; } return 'idle'; } function getBattlePresentationPoses(event, phase) { const poses = { player: 'idle', opponent: 'idle' }; if (!event) { return poses; } const actor = event.actor === 'opponent' ? 'opponent' : 'player'; const target = actor === 'player' ? 'opponent' : 'player'; if (phase === 'guard') { poses[actor] = 'guard'; } else if (phase === 'attack' || phase === 'impact') { poses[actor] = 'attack'; } if (phase === 'impact' || phase === 'hurt') { poses[target] = 'hurt'; } return poses; } function getBattlePhaseProgress(phase, localElapsed) { const ranges = { attack: [0, 190], impact: [190, 310], hurt: [310, 520], guard: [0, 420] }; const range = ranges[phase]; if (!range) { return 1; } return Math.max(0, Math.min(1, (localElapsed - range[0]) / Math.max(1, range[1] - range[0]))); } function getBattlePresentationOffset(presentation, actor, now) { const phase = presentation && presentation.phase || 'idle'; const event = presentation && presentation.event; const isActor = event && (event.actor === 'opponent' ? 'opponent' : 'player') === actor; const isTarget = event && !isActor; if ((phase === 'attack' || phase === 'impact') && isActor) { const direction = actor === 'player' ? 1 : -1; return { x: Math.round(direction * (5 + presentation.phaseProgress * 9)), y: -2 }; } if ((phase === 'impact' || phase === 'hurt') && isTarget) { return { x: Math.round(Math.sin((Number(now) || 0) / 27) * 4), y: 2 }; } if (phase === 'guard' && isActor) { return { x: 0, y: 3 }; } return { x: 0, y: 0 }; } function drawBattleImpactEffects(ctx, presentation) { const event = presentation && presentation.event; if (!event || event.type !== 'hit' || presentation.phase !== 'impact') { return; } const targetX = event.actor === 'player' ? 360 : 154; const targetY = 276; const spread = 10 + Math.round((Number(presentation.phaseProgress) || 0) * 18); const shards = [ [-spread, -spread, 10, 4], [spread - 8, -spread + 4, 8, 5], [-spread - 4, spread - 5, 7, 5], [spread, spread - 9, 9, 4], [-5, -spread - 8, 6, 9], [3, spread + 2, 5, 8] ]; ctx.save(); ctx.globalAlpha = 0.92 - (Number(presentation.phaseProgress) || 0) * 0.32; shards.forEach(function(shard, index) { ctx.fillStyle = index % 2 === 0 ? '#fff6b8' : '#d8646f'; ctx.fillRect(targetX + shard[0], targetY + shard[1], shard[2], shard[3]); }); ctx.restore(); }
