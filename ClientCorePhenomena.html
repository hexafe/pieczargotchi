0.42 : 1), seed + 211, 0.16 ); const fogbow = createAmbientPhenomenon('fogbow', fogPotential * lowSun * dayLight * noHeavyPrecip * clampNumber(1 - lowCloud / 160, 0.32, 1, 1), seed + 307, 0.2 ); const moonHalo = createAmbientPhenomenon('moonHalo', night * highCloudHalo * noPrecip * (condition === 'storm' || condition === 'snow' ? 1 : 0) * getPhenomenaWindow(hour, 6.4, 8.4, 15.5, 18.4), seed + 607, 0.18 ); const heatHaze = createAmbientPhenomenon('heatHaze', hotFactor * dryAir * noon * noPrecip * clampNumber(1 - wetness * 0.88, 0, 1, 1) * (condition === 'clear' || condition === 'cloudy' ? 1 : 0), seed + 701, 0.2 ); const clearingAfterRain = createAmbientPhenomenon('clearingAfterRain', recentRain * dayLight * noPrecip * (clearingWeather ? 1 : 0) * clampNumber(0.42 + cloudGap * 0.58, 0, 1, 0), seed + 809, 0.24 ); const discoveries = [dew, frost, fogbow, moonHalo, sunbeams, steam, heatHaze, clearingAfterRain] .filter(function(item) { return item.discoverable; }) .map(function(item) { return item.id; }); return { dew: dew, frost: frost, fogbow: fogbow, moonHalo: moonHalo, sunbeams: sunbeams, steam: steam, heatHaze: heatHaze, clearingAfterRain: clearingAfterRain, discoveries: discoveries }; } function getEnvironmentDiscoveryCatalog() { return environmentDiscoveryCatalog.map(function(item) { return Object.assign({}, item); }); } function recordEnvironmentDiscovery(rawState, discoveryId, nowInput) { const state = isPlainObject(rawState) ? 1 : 0; } function getPhenomenaNightFactor(scene, hour) { if (scene && (scene.isDay === false || scene.dayPhase === 'night')) { return 1; } if (scene && scene.dayPhase === 'sunset') { return 0.32; } if (hour >= 21 || hour < 4.5) { return 1; } return 0; } function getPhenomenaLowSunFactor(scene, hour) { if (scene && (scene.dayPhase === 'sunrise' || scene.dayPhase === 'sunset')) { return 1; } if (scene && (scene.dayTone === 'dawnGold' || scene.dayTone === 'duskGold' || scene.dayTone === 'dawnBlue' || scene.dayTone === 'duskBlue')) { return 0.82; } return Math.max( getPhenomenaWindow(hour, 4.9, 5.6, 7.1, 8.2), getPhenomenaWindow(hour, 17.4, 18.2, 20.1, 21.0) ); } function getPhenomenaCloudGapFactor(cloudCover, lowCloud) { const total = clampNumber(Number(cloudCover), 0, 100, 30); const low = clampNumber(Number(lowCloud), 0, 100, total); const enoughCloud = smoothPhenomenaStep(16, 46, total); const notOvercast = 1 - smoothPhenomenaStep(84, 100, total); const notLowWall = 1 - smoothPhenomenaStep(76, 100, low); return clampNumber(enoughCloud * notOvercast * notLowWall, 0, 1, 0); } function getPhenomenaWindow(hour, start, peakStart, peakEnd, end) { return smoothPhenomenaStep(start, peakStart, hour) * (1 - smoothPhenomenaStep(peakEnd, end, hour)); } function smoothPhenomenaStep(edge0, edge1, value) { const t = clampNumber((value - edge0) / Math.max(0.0001, edge1 - edge0), 0, 1, 0); return t * t * (3 - 2 * t); } function getPhenomenaDaySeed(date) { return date.getFullYear() * 379 + (date.getMonth() + 1) * 37 + date.getDate(); } function phenomenaSeededUnit(seed) { let value = Math.imul(Math.floor(seed) ^ 0x9e3779b9, 0x85ebca6b); value ^= value >>> 13; value = Math.imul(value, 0xc2b2ae35); value ^= value >>> 16; return (value >>> 0) / 4294967295; } function roundPhenomenaUnit(value) { return Math.round(clampNumber(value, 0, 1, 0) * 1000) / 1000; }
