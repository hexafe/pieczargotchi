const IMMERSION_DEFAULTS = { pointerFreshMs: 1800, pointerTapFreshMs: 1300, pointerFaceRadius: 170, pointerGroundRadius: 96, pointerMinSpeed: 0.42, pointerFastMinSpeed: 0.86, pointerTrailFreshMs: 1250, pointerHoverDurationMs: 1500, pointerHoverCooldownMs: 900, pointerTapDurationMs: 1150, pointerTapCooldownMs: 2400, weatherDurationMs: 2600, weatherCooldownMs: 8200, celestialDurationMs: 3000, celestialCooldownMs: 10000, windDurationMs: 1800, windCooldownMs: 6200, ambientLifeDurationMs: 1900, ambientLifeCooldownMs: 7200, idleFidgetDurationMs: 1700, idleFidgetCooldownMs: 8800 }; function selectImmersionReaction(state, scene, input, now, rules) { const timestamp = Number(now) || 0; const config = Object.assign({}, IMMERSION_DEFAULTS, rules && rules.immersion ? rules.immersion : {}); } return null; } if (isPointNearMushroomCore(x, y, 'face', config.pointerFaceRadius)) { const state = selectPointerLookState(input, 'hover'); return buildImmersionCandidate('pointerHover', state, 'input', movedAt, config.pointerHoverDurationMs, config.pointerHoverCooldownMs, 80, 'watch_cursor', 'pointer'); } if (speed >= config.pointerMinSpeed && isPointNearMushroomCore(x, y, 'ground', config.pointerGroundRadius)) { return buildImmersionCandidate('pointerGrass', 'follow_cursor_after', 'input', movedAt, 980, 900, 74, 'follow_cursor', 'pointer'); } return null; } function selectPointerLookState(input, mode) { const rawX = readOptionalCoreNumber(input.x); const rawY = readOptionalCoreNumber(input.y); const x = rawX === null ? 256 : rawX; const y = rawY === null ? 256 : rawY; const previousX = readOptionalCoreNumber(input.previousX); const dx = x - 256; const dy = y - 268; const movedRight = previousX !== null ? x >= previousX : dx >= 0; const lookRight = Math.abs(dx) < 18 ? movedRight : dx >= 0; if (mode === 'tap' && dy > 78) { return lookRight ? 'watch_cursor_right' : 'watch_cursor_left'; } if (dy < -28) { return lookRight ? 'watch_cursor_up_right' : 'watch_cursor_up_left'; } return lookRight ? 'watch_cursor_right' : 'watch_cursor_left'; } function isPointerPathNearMushroomCore(input, radius) { const x = readOptionalCoreNumber(input.x); const y = readOptionalCoreNumber(input.y); const previousX = readOptionalCoreNumber(input.previousX); const previousY = readOptionalCoreNumber(input.previousY); if (x === null || y === null || previousX === null || previousY === null) { return false; } const center = { x: 256, y: 296 }; const vx = x - previousX; const vy = y - previousY; const lenSq = vx * vx + vy * vy; if (lenSq <= 0.001) { return isPointNearMushroomCore(x, y, 'body', radius); } const t = clampNumber(((center.x - previousX) * vx + (center.y - previousY) * vy) / lenSq, 0, 1, 0); const closestX = previousX + vx * t; const closestY = previousY + vy * t; return Math.hypot(closestX - center.x, closestY - center.y) <= radius; } function selectWeatherImmersionReaction(scene, now, config) { if (!isPlainObject(scene)) { return null; } const condition = scene.condition || 'clear'; const precipitation = Math.max( 0, Number(scene.precipitation) || 0, Number(scene.rain) || 0, Number(scene.showers) || 0, Number(scene.snowfall) || 0 ); if (condition === 'snow' || Number(scene.snowfall) > 0.04 || Number(scene.snowIntensity) > 0.05) { return buildImmersionCandidate('snow', 'snow', 'weather', now, config.weatherDurationMs + 400, config.weatherCooldownMs, 64); } if (['rain', 'storm'].includes(condition) && precipitation > 0.04) { return buildImmersionCandidate('rain', 'rain', 'weather', now, config.weatherDurationMs, config.weatherCooldownMs, condition === 'storm' ? 66 : 62); } const windLevel = Math.max(0, Number(scene.windLevel) || 0, Number(scene.gustLevel) || 0); if (windLevel >= 0.72 && precipitation <= 0.04) { return buildImmersionCandidate('wind', 'curious', 'weather', now, config.windDurationMs, config.windCooldownMs, 48); } return null; } function selectCelestialImmersionReaction(scene, now, config) { if (!isPlainObject(scene)) { return null; } const cloudCover = clampNumber(Number(scene.cloudCover), 0, 100, 0); const isDay = Boolean(scene.isDay); const condition = scene.condition || 'clear'; const precipitation = Math.max(0, Number(scene.precipitation) || 0, Number(scene.rain) || 0, Number(scene.snowfall) || 0); if (isDay && !['rain', 'storm', 'snow', 'fog'].includes(condition) && precipitation <= 0.03 && cloudCover <= 62) { return buildImmersionCandidate('sun', 'sun', 'celestial', now, config.celestialDurationMs - 400, config.celestialCooldownMs, 42); } const starVisibility = Number(scene.starVisibility); const visibleStars = Number.isFinite(starVisibility) ? starVisibility > 0.12 : (!isDay && cloudCover <= 68 && !['rain', 'storm', 'snow', 'fog'].includes(condition)); if (!isDay && visibleStars && precipitation <= 0.03) { return buildImmersionCandidate('stargaze', 'stargaze', 'celestial', now, config.celestialDurationMs, config.celestialCooldownMs, 44); } return null; } function selectAmbientLifeImmersionReaction(scene, now, config) { if (typeof calculateAmbientLifeFocusCue !== 'function') { return null; } const cue = calculateAmbientLifeFocusCue(scene, new Date(now), now); if (!cue || !cue.visible || !cue.state) { return null; } const idByKind = { butterfly: 'ambientButterfly', firefly: 'ambientFirefly', crawler: 'ambientCrawler' }; const priorityByKind = { butterfly: 57, firefly: 56, crawler: 55 }; const candidate = buildImmersionCandidate( idByKind[cue.kind] || 'ambientLife', cue.state, 'ambient', cue.startedAt || now, config.ambientLifeDurationMs, config.ambientLifeCooldownMs, priorityByKind[cue.kind] || 54 ); candidate.cue = cue; return candidate; } function selectIdleImmersionReaction(scene, input, now, config) { const source = isPlainObject(scene) ? 'idlePonder' : 'idleFidget', state, 'idle', cycleStart, config.idleFidgetDurationMs, config.idleFidgetCooldownMs, 38, group, group ); } function getImmersionVariants(group) { const variants = { idle_fidget: ['idle_fidget_sway', 'idle_fidget_shift', 'idle_look_left', 'idle_look_right'], ponder: ['ponder_up', 'ponder_side', 'ponder_breath'] }; return variants[group] || []; } function chooseImmersionVariant(group, variants, seed, input) { if (!variants.length) { return group; } const lastByGroup = isPlainObject(input) && isPlainObject(input.lastVariantByGroup) ? input.lastVariantByGroup : {}; const last = lastByGroup[group] || ''; let index = Math.floor(immersionSeededUnit(seed) * variants.length) % variants.length; if (variants.length > 1 && variants[index] === last) { index = (index + 1 + Math.floor(immersionSeededUnit(seed + 7) * (variants.length - 1))) % variants.length; } return variants[index]; } function buildImmersionCandidate(id, state, source, sourceAt, durationMs, cooldownMs, priority, variantGroup, logicalState) { return { id: id, state: state, logicalState: logicalState || state, variantGroup: variantGroup || '', source: source, sourceAt: sourceAt, durationMs: Math.max(300, Number(durationMs) || 1200), cooldownMs: Math.max(0, Number(cooldownMs) || 0), priority: Number(priority) || 0 }; } function isPointNearMushroomCore(x, y, zone, radius) { const centers = { face: { x: 256, y: 260, rx: radius, ry: radius * 0.72 }, body: { x: 256, y: 300, rx: radius, ry: radius * 0.95 }, ground: { x: 256, y: 424, rx: radius * 2.4, ry: radius * 0.75 } }; const center = centers[zone] || centers.body; const dx = (Number(x) - center.x) / center.rx; const dy = (Number(y) - center.y) / center.ry; return dx * dx + dy * dy <= 1; } function immersionSeededUnit(seed) { let value = Math.imul(Math.floor(seed) ^ 0x9e3779b9, 0x85ebca6b); value ^= value >>> 13; value = Math.imul(value, 0xc2b2ae35); value ^= value >>> 16; return (value >>> 0) / 4294967295; }
