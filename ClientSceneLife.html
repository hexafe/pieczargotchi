function drawAmbientLifeBackground(ctx, palette, now, scene) { const profile = getAmbientLifeProfile(scene); if (!hasAmbientLife(profile)) { return; } drawButterflies(ctx, palette, now, scene, profile); drawSmallFlyingInsects(ctx, palette, now, scene, profile); drawFireflies(ctx, palette, now, scene, profile, false); } function drawAmbientLifeForeground(ctx, now) { const scene = getCurrentWeatherScene(); const palette = getScenePalette(scene); const profile = getAmbientLifeProfile(scene); if (!hasAmbientLife(profile)) { return; } drawButterflies(ctx, palette, now, scene, profile, true); drawSmallFlyingInsects(ctx, palette, now, scene, profile, true); drawCrawlingBugs(ctx, palette, now, scene, profile); drawFireflies(ctx, palette, now, scene, profile, true); } function getAmbientLifeProfile(scene) { const core = window.PieczargotchiCore; if (core && typeof core.calculateAmbientLife === 'function') { return core.calculateAmbientLife(scene, getRuntimeDate()); } const isDay = scene && scene.isDay !== false; const condition = scene && scene.condition ? scene.condition : 'clear'; const weatherFactor = condition === 'clear' || condition === 'cloudy' ? 1 : 0; return { generalIntensity: isDay ? weatherFactor * 0.42 : 0.02, butterflyIntensity: isDay ? weatherFactor * 0.28 : 0, flyingInsectIntensity: isDay ? weatherFactor * 0.32 : 0.06, crawlerIntensity: weatherFactor * 0.22, fireflyIntensity: 0 }; } function hasAmbientLife(profile) { return Boolean(profile) && ( Number(profile.butterflyIntensity) > 0.015 || Number(profile.flyingInsectIntensity) > 0.015 || Number(profile.crawlerIntensity) > 0.015 || Number(profile.fireflyIntensity) > 0.015 ); } function drawButterflies(ctx, palette, now, scene, profile, foreground) { const intensity = clamp(Number(profile.butterflyIntensity), 0, 1, 0); if (intensity < 0.045) { return; } const wind = getDynamicWindState(scene, now); const slots = Math.max(2, Math.round(2 + intensity * 4)); const colors = [ { wing: '#f4c45f', edge: '#8c6731', body: '#3d2c22' }, { wing: '#d98956', edge: '#7a4a33', body: '#33251e' }, { wing: '#b6d96b', edge: '#4f7b39', body: '#2f3b22' } ]; ctx.save(); for (let index = 0; index < slots; index += 1) { const seed = 18011 + index * 173; if (intensity < 0.9 && seededUnit(seed + 17) > intensity + 0.28) { continue; } const cycleMs = 11800 + seededUnit(seed) * 9400; const activeSpan = 0.46 + intensity * 0.32; const phase = positiveModulo(now + seededUnit(seed + 1) * cycleMs, cycleMs) / cycleMs; if (phase > activeSpan) { continue; } const progress = phase / activeSpan; const direction = seededUnit(seed + 2) > 0.5 ? 1 : -1; const travel = direction > 0 ? -36 + progress * 584 : 548 - progress * 584; const yBase = 194 + seededUnit(seed + 3) * 126; const y = yBase + Math.sin(progress * Math.PI * 2.2 + seed) * (12 + seededUnit(seed + 4) * 18) + Math.sin(now / (620 + seededUnit(seed + 5) * 260) + seed) * 4 + wind.vectorY * wind.level * 18; const x = travel + wind.directionX * wind.level * 22 * Math.sin(progress * Math.PI); const size = 4 + Math.round(seededUnit(seed + 6) * 2 + intensity * 1.8); const flap = (Math.sin(now / (92 + seededUnit(seed + 7) * 48) + seed) + 1) / 2; const alpha = (0.58 + intensity * 0.34) * fadeLifeEdge(progress); if (foreground && isMushroomLifeOccluded(x, y)) { continue; } ctx.globalAlpha = foreground ? Math.min(1, alpha * 1.08) : alpha; drawPixelButterfly(ctx, Math.round(x), Math.round(y), direction, size, flap, colors[index % colors.length]); } ctx.restore(); } function drawPixelButterfly(ctx, x, y, direction, size, flap, colors) { const wingLift = Math.round(1 + flap * 3); const wingWide = size + Math.round(flap * 2); const wingHigh = size + 2; const bodyHeight = Math.max(3, size + 1); ctx.fillStyle = colors.edge; ctx.fillRect(x - direction * (wingWide + 3), y - wingLift, wingWide, wingHigh); ctx.fillRect(x + direction * 3, y - wingLift + 1, wingWide, wingHigh - 1); ctx.fillStyle = colors.wing; ctx.fillRect(x - direction * (wingWide + 2), y - wingLift + 1, Math.max(2, wingWide - 2), Math.max(2, wingHigh - 2)); ctx.fillRect(x + direction * 4, y - wingLift + 2, Math.max(2, wingWide - 2), Math.max(1, wingHigh - 3)); ctx.fillStyle = colors.body; ctx.fillRect(x - 1, y - 1, 2, bodyHeight); ctx.fillRect(x + direction * 1, y - 2, 2, 2); } function drawSmallFlyingInsects(ctx, palette, now, scene, profile, foreground) { const intensity = clamp(Number(profile.flyingInsectIntensity), 0, 1, 0); if (intensity < 0.035) { return; } const wind = getDynamicWindState(scene, now); const slots = Math.round(2 + intensity * 8); ctx.save(); ctx.fillStyle = palette.ground === '#d9e5cf' ? '#6e7e67' : '#2e3a22'; for (let index = 0; index < slots; index += 1) { const seed = 21017 + index * 149; if (seededUnit(seed + 8) > intensity + 0.38) { continue; } const cycleMs = 4200 + seededUnit(seed) * 5200; const phase = positiveModulo(now + seededUnit(seed + 1) * cycleMs, cycleMs) / cycleMs; const drift = Math.sin(now / (850 + seededUnit(seed + 2) * 700) + seed); const x = 48 + seededUnit(seed + 3) * 416 + Math.sin(phase * Math.PI * 2 + seed) * (12 + seededUnit(seed + 4) * 28) + wind.directionX * wind.level * 16; const y = 238 + seededUnit(seed + 5) * 164 + drift * (4 + seededUnit(seed + 6) * 11) + wind.vectorY * wind.level * 10; const twinkle = 0.48 + ((Math.sin(now / 140 + seed) + 1) / 2) * 0.34; if (foreground && isMushroomLifeOccluded(x, y)) { continue; } ctx.globalAlpha = (foreground ? 1.18 : 1) * (0.22 + intensity * 0.32) * twinkle; ctx.fillRect(Math.round(x), Math.round(y), 2, 2); if (seededUnit(seed + 7) > 0.58) { ctx.fillRect(Math.round(x + 3), Math.round(y - 1), 1, 1); } } ctx.restore(); } function drawCrawlingBugs(ctx, palette, now, scene, profile) { const intensity = clamp(Number(profile.crawlerIntensity), 0, 1, 0); if (intensity < 0.04) { return; } const wind = getDynamicWindState(scene, now); const lanes = [ { from: -24, to: 146, y: 462 }, { from: 540, to: 358, y: 468 }, { from: 24, to: 214, y: 480 }, { from: 496, to: 304, y: 452 } ]; const slots = Math.round(2 + intensity * 5); ctx.save(); for (let index = 0; index < slots; index += 1) { const seed = 24019 + index * 181; if (seededUnit(seed + 9) > intensity + 0.24) { continue; } const lane = lanes[index % lanes.length]; const cycleMs = 16000 + seededUnit(seed) * 18000 - intensity * 5200; const activeSpan = 0.68 + intensity * 0.24; const phase = positiveModulo(now + seededUnit(seed + 1) * cycleMs, cycleMs) / cycleMs; if (phase > activeSpan) { continue; } const progress = phase / activeSpan; const eased = easeLifeProgress(progress); const x = lane.from + (lane.to - lane.from) * eased + Math.sin(now / 740 + seed) * 2; const y = lane.y + Math.sin(progress * Math.PI * 4 + seed) * 1.5 + wind.vectorY * wind.level * 4; const direction = lane.to >= lane.from ? 1 : -1; const size = 5 + Math.round(seededUnit(seed + 2) * 3); const alpha = (0.55 + intensity * 0.34) * fadeLifeEdge(progress); ctx.globalAlpha = alpha; drawPixelCrawler(ctx, Math.round(x), Math.round(y), direction, size, now, palette, seed); } ctx.restore(); } function drawPixelCrawler(ctx, x, y, direction, size, now, palette, seed) { const body = seededUnit(seed + 3) > 0.44 ? '#6a3b20' : '#31451f'; const outline = '#1f1915'; const highlight = palette.groundLight || '#83b955'; const legPhase = Math.sin(now / 120 + seed) >= 0 ? 1 : -1; const length = size + 7; const height = Math.max(4, size); ctx.fillStyle = outline; ctx.fillRect(x - 1, y - 3, length + 2, height + 2); ctx.fillStyle = body; ctx.fillRect(x, y - 2, length, height); ctx.fillRect(x + direction * (length - 1), y - 3, 3, 3); ctx.fillStyle = highlight; ctx.globalAlpha *= 0.62; ctx.fillRect(x + 2, y - 1, 2, 1); ctx.globalAlpha /= 0.62; ctx.fillStyle = outline; for (let leg = 0; leg < 3; leg += 1) { const legX = x + 2 + leg * 3; const legOffset = leg % 2 === 0 ? legPhase : -legPhase; ctx.fillRect(legX, y + size - 1, 1, 2 + legOffset); ctx.fillRect(legX + direction * 1, y - 4 - legOffset, 1, 2); } } function drawFireflies(ctx, palette, now, scene, profile, foreground) { const intensity = clamp(Number(profile.fireflyIntensity), 0, 1, 0); if (intensity < 0.035) { return; } const wind = getDynamicWindState(scene, now); const slots = foreground ? Math.round(3 + intensity * 7) : Math.round(5 + intensity * 10); const yMin = foreground ? 356 : 292; const yRange = foreground ? 94 : 118; ctx.save(); for (let index = 0; index < slots; index += 1) { const seed = (foreground ? 29033 : 27031) + index * 131; if (seededUnit(seed + 11) > intensity + 0.44) { continue; } const baseX = 36 + seededUnit(seed) * 440; const baseY = yMin + seededUnit(seed + 1) * yRange; const flutterX = Math.sin(now / (1700 + seededUnit(seed + 2) * 1400) + seed) * (8 + intensity * 12); const flutterY = Math.sin(now / (2100 + seededUnit(seed + 3) * 1300) + seed * 0.7) * (5 + intensity * 9); const x = baseX + flutterX + wind.directionX * wind.level * 13; const y = baseY + flutterY + wind.vectorY * wind.level * 8; const pulse = (Math.sin(now / (480 + seededUnit(seed + 4) * 520) + seed) + 1) / 2; const glow = smoothStep(0.34, 0.98, pulse) * (0.48 + intensity * 0.52); const alpha = glow * (foreground ? 0.86 : 0.72); if (alpha < 0.06) { continue; } drawPixelFirefly(ctx, Math.round(x), Math.round(y), alpha, foreground); } ctx.restore(); } function drawPixelFirefly(ctx, x, y, alpha, foreground) { const glowSize = foreground ? 9 : 7; ctx.globalAlpha = alpha * 0.22; ctx.fillStyle = '#f1ff9d'; ctx.fillRect(x - Math.floor(glowSize / 2), y - Math.floor(glowSize / 2), glowSize, glowSize); ctx.globalAlpha = alpha * 0.62; ctx.fillStyle = '#dfff6d'; ctx.fillRect(x - 2, y - 1, 4, 3); ctx.globalAlpha = alpha; ctx.fillStyle = '#fff7b2'; ctx.fillRect(x, y, 2, 2); } function fadeLifeEdge(progress) { return clamp(Math.min(progress / 0.12, (1 - progress) / 0.14), 0, 1, 0); } function easeLifeProgress(progress) { return 0.5 - Math.cos(clamp(progress, 0, 1, 0) * Math.PI) / 2; } function isMushroomLifeOccluded(x, y) { return x > 152 && x < 362 && y > 198 && y < 398; }
